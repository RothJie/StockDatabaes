import json
from DB import DB

table_names = DB.executeUniSQL("show tables")

fi = open(file='sql_s.json', mode='w', encoding="utf8")

sql_s = []
for table_name in table_names:
    sql = f"""select 股票代码,股票名称,所属行业,最高,最低,收盘,成交量 from {table_name} WHERE 股票代码 = '{table_name.rstrip("csv")}' AND NOT 成交量='-' limit 360;"""
    sql_s.append(sql)
json.dump(obj=sql_s, fp=fi, indent=4, ensure_ascii=False)
fi.close()
